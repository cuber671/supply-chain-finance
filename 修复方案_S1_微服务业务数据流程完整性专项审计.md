### 1.2 修复原则

1. **区块链优先原则**: 涉及区块链上链的操作，必须先执行区块链调用，成功后更新数据库
2. **Fail-Fast原则**: 数据校验失败时立即抛出异常，不使用静默fallback
3. **事务一致性原则**: 区块链失败时抛出异常触发事务回滚，保持链上链下一致

@Scheduled(cron = "0 0 2 * * ?")  // 每天凌晨2点重试
public void retryPendingCreditDeductions() {
    LambdaQueryWrapper<LogisticsDeviationCreditRecord> wrapper =
        new LambdaQueryWrapper<>();
    wrapper.eq(LogisticsDeviationCreditRecord::getStatus, 0)  // PENDING
          .lt(LogisticsDeviationCreditRecord::getRetryCount, 3);  // 重试不超过3次

    List<LogisticsDeviationCreditRecord> pendingRecords =
        deviationCreditRecordMapper.selectList(wrapper);

    for (LogisticsDeviationCreditRecord record : pendingRecords) {
        try {
            CreditFeignClient.LogisticsDeviationRequest deviationRequest =
                new CreditFeignClient.LogisticsDeviationRequest();
            deviationRequest.setEntId(record.getEntId());
            deviationRequest.setLogisticsOrderId(record.getLogisticsOrderId());
            deviationRequest.setDeviationLevel(record.getDeviationLevel());
            deviationRequest.setDeviationDesc(record.getDeviationDesc());

            Map<String, Object> result = creditFeignClient.reportLogisticsDeviation(deviationRequest);
            if (result != null && "0".equals(String.valueOf(result.get("code"))))) {
                record.setStatus(1);  // SUCCESS
                logger.info("偏航信用扣分重试成功: entId={}, voucherNo={}",
                    record.getEntId(), record.getLogisticsOrderId());
            } else {
                throw new RuntimeException("重试返回失败");
            }
        } catch (Exception e) {
            record.setRetryCount(record.getRetryCount() + 1);
            record.setLastRetryTime(LocalDateTime.now());
            record.setErrorMsg(e.getMessage());
            if (record.getRetryCount() >= 3) {
                record.setStatus(2);  // FAILED
                logger.error("偏航信用扣分重试失败已达上限: entId={}, voucherNo={}",
                    record.getEntId(), record.getLogisticsOrderId());
            }
        }
        deviationCreditRecordMapper.updateById(record);
    }
}
